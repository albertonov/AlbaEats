CREATE OR REPLACE TYPE RESTAURANTE_OBJ AS OBJECT(
ID_RESTAURANTE NUMBER(10,0),
NOMBRE VARCHAR2(20),
DIRECCION VARCHAR2(200),
CIUDAD VARCHAR(30),
CODIGO_POSTA NUMBER(5),
TELEFONO NUMBER(9),
TIPORESTAURANTE VARCHAR(30),
HORA_APERTURA TIMESTAMP (1),
HORA_CIERRE TIMESTAMP(1),
CALIFICACION NUMBER(1),
PRODUCTO REF RESTAURANTE_OBJ
);

CREATE OR REPLACE TYPE PRODUCTO_OBJ AS OBJECT(
ID_PRODUCTO NUMBER(10),
NOMBRE VARCHAR(20),
DESCRIPCION VARCHAR2(200),
PRECIO_UNIT NUMBER(6,2),
STOCK NUMBER(6),
PESO_GRAMOS NUMBER(3),
CALORIAS NUMBER(3)
);

CREATE OR REPLACE TYPE LINEAPEDIDO_OBJ AS OBJECT(
ID_LPEDIDO NUMBER(6),
CANTIDAD NUMBER(2),
PRECIO NUMBER(6,2),
PRODUCTO REF PRODUCTO_OBJ
);

CREATE TYPE LINEAPEDIDO_nTAB AS TABLE OF LINEAPEDIDO_OBJ;

CREATE OR REPLACE TYPE PEDIDO_OBJ AS OBJECT(
ID_PEDIDO NUMBER(6),
PRECIO_TOTAL NUMBER(6),
DISTANCIA NUMBER(5,2),
FECHA DATE,
PAGADO NUMBER(1),
URGENCIA NUMBER(1),
TIEMPO_ENTREGA TIMESTAMP(1),
REPARTIDOR REF REPARTIDOR_OBJ,
FACTURA REF FACTURA_OBJ
);

CREATE OR REPLACE TYPE CLIENTE_OBJ AS OBJECT(
ID_CLIENTE NUMBER(10),
NOMBRE VARCHAR2(20),
APELLIDOS VARCHAR2(40),
DIRECCION VARCHAR2(100),
CODIGO_POSTAL NUMBER(5),
TELEFONO NUMBER(9),
CORREO_E VARCHAR2(75)
);

CREATE OR REPLACE TYPE TARJETACREDITO_OBJ AS OBJECT(
NUMERO NUMBER(16),
FECHA_CADUCIDAD NUMBER(4),
CVV NUMBER(3),
PROPIETARIO VARCHAR2(50),
CLIENTE CLIENTE_OBJ
);
